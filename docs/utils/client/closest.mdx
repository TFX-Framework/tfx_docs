#### Example
```lua
Util.Tick(function()
	local ply,dist = Util.GetClosestPlayer()
	if dist < 5 then 
		local pos = GetEntityCoords(GetPlayerPed(ply))
		Util.DrawText3D(pos.x, pos.y, pos.z 'Player')
	end
end)
``` 
